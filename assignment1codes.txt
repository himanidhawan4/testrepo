1.top sale
db.Tickets.aggregate([
  { $match: { status: "booked" } },
  { 
    $lookup: {
      from: "Events",
      localField: "eventId",
      foreignField: "eventId",
      as: "eventInfo"
    } 
  },
  { $unwind: "$eventInfo" },
  { $match: { "eventInfo.organizerId": "U1011" } }, // Filter by organizer here
  { 
    $group: { 
      _id: "$eventInfo.organizerId", 
      totalRevenue: { $sum: "$totalAmount" } 
    } 
  }
])
2.revenue
db.Events.aggregate([
  { $match: { organizerId: "U1011" } }, 
  { $lookup: {
      from: "Tickets",
      localField: "eventId",
      foreignField: "eventId",
      as: "bookings"
  }},
  { $unwind: "$bookings" },
  { $match: { "bookings.status": "booked" } },
  { $group: { 
      _id: "$organizerId", 
      totalRevenue: { $sum: "$bookings.totalAmount" } 
  }}
])
